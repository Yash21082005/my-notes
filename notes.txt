package com.ltfs.brake.collections.datasource;

public class BrakeDataSourceContext {

    private static final ThreadLocal<BrakeDataSourceType> CONTEXT =
            new ThreadLocal<>();

    public static void set(BrakeDataSourceType type) {
        CONTEXT.set(type);
    }

    public static BrakeDataSourceType get() {
        return CONTEXT.get();
    }

    public static void clear() {
        CONTEXT.remove();
    }
}
