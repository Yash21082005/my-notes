package com.ltfs.brake.collections.datasource;

public class BrakeDataSourceContext {

    private static final ThreadLocal<BrakeDataSourceType> CONTEXT =
            new ThreadLocal<>();

    public static void setDataSource(BrakeDataSourceType dataSourceType) {
        CONTEXT.set(dataSourceType);
    }

    public static BrakeDataSourceType getDataSource() {
        return CONTEXT.get();
    }

    public static void clear() {
        CONTEXT.remove();
    }
}
